function doPost(e) {
  var d = JSON.parse(e.postData.contents);
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sh = ss.getSheetByName('2026 Transaction Log')
    || ss.getSheets()[0];

  sh.appendRow([
    d.date, d.item, d.type, d.category,
    d.supplier, d.taxId, d.address,
    d.priceExVat, d.taxPct, d.taxAmount,
    d.whtPct, d.whtAmount, d.total,
    d.netAmount, d.invoiceNo, d.project,
    d.account, d.notes, d.pndType
  ]);

  return ContentService
    .createTextOutput(JSON.stringify({ok:true}))
    .setMimeType(ContentService.MimeType.JSON);
}
